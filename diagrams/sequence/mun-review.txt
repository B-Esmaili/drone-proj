@startuml
actor Municipal
participant ProjectApprovalSystem
participant Admin
participant Pilot

Municipal -> ProjectApprovalSystem: Review Project
activate ProjectApprovalSystem

alt Project Approved
    ProjectApprovalSystem -> ProjectApprovalSystem: Forward to Pilot
    ProjectApprovalSystem -> Pilot: Forward to Pilot
else Project Rejected
    ProjectApprovalSystem -> Admin: Send back with Specific Message
end

deactivate ProjectApprovalSystem
@enduml
