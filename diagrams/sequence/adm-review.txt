@startuml
actor Admin
participant ProjectApprovalSystem
participant Employer
participant Municipal

Admin -> ProjectApprovalSystem: Review Project
activate ProjectApprovalSystem

alt Project Rejected
    ProjectApprovalSystem -> Employer: Send back with Message
else Project Approved
    ProjectApprovalSystem -> ProjectApprovalSystem: Forward to Municipal
    ProjectApprovalSystem -> Municipal: Forward to Municipal
end

deactivate ProjectApprovalSystem
@enduml
